# quick sort
```java
sort(int[]a, int lo, int hi){
    int j = partition(a, lo, hi);
    sort(a, lo, j-1);
    sort(a, j+1, hi);
}


private int partition(int[] a, int lo, int hi){
    int i = lo, j = hi+1;
    while(true){
        while(a[++i] < a[lo])
            if (i == hi) break;
        while(a[lo] < a[--j])
            if (j == lo) break;
        if(i >= j) break;
        exchange(a, i, j);
    }
    exchange(a, lo, j);
    return j;
}
```
